<EndpointPath type="query" path="get_session_profiles" />

<Authorizations />

<H3Bordered text="Body" />

<ParamField body="organizationId" type="string" required={true}>

Unique identifier for a given organization.
</ParamField>


<H3Bordered text="Response" />
A successful response returns the following fields:

<ResponseField name="sessionProfiles" type="array" required={true}>
  A list of session profiles for users in the organization.
  <Expandable title="sessionProfiles details">
    <NestedParam parentKey="sessionProfiles" childKey="sessionProfileId" type="string" required={true}>
Unique identifier for a given Session Profile.
</NestedParam>
<NestedParam parentKey="sessionProfiles" childKey="sessionProfileName" type="string" required={true}>
Human-readable name for a Session Profile.
</NestedParam>
<NestedParam parentKey="sessionProfiles" childKey="scope" type="string" required={true}>
The specific scope that a session created with this profile is limited to.
</NestedParam>
<NestedParam parentKey="sessionProfiles" childKey="expirationSeconds" type="string" required={false}>
Optional window (in seconds) indicating how long sessions created with this profile should last.
</NestedParam>
<NestedParam parentKey="sessionProfiles" childKey="notes" type="string" required={false}>
Optional human-readable notes added by a User to describe a particular Session Profile.
</NestedParam>
<NestedParam parentKey="sessionProfiles" childKey="createdAt" type="object" required={true}>
      createdAt field

nanos field
</NestedParam>

      </Expandable>
    </NestedParam>
<NestedParam parentKey="sessionProfiles" childKey="updatedAt" type="object" required={true}>
      updatedAt field

    </NestedParam>

  </Expandable>
</ResponseField>
