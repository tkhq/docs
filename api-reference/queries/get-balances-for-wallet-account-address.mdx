---

import { EndpointPath } from "/snippets/api/endpoint.mdx";

<EndpointPath type="query" path="get_wallet_address_balances" />

<Authorizations />

<H3Bordered text="Body" />

<ParamField body="organizationId" type="string" required={true}>

Unique identifier for a given organization.
</ParamField>

<ParamField body="address" type="string" required={true}>

Address corresponding to a wallet account.
</ParamField>

<ParamField body="caip2" type="string" required={true}>

The network identifier in CAIP-2 format (e.g., 'eip155:1' for Ethereum mainnet).
</ParamField>


<H3Bordered text="Response" />
A successful response returns the following fields:

<ResponseField name="balances" type="array" required={false}>
  balances field
  <Expandable title="balances details">
    <NestedParam parentKey="balances" childKey="caip19" type="string" required={false}>
caip19 field
</NestedParam>
<NestedParam parentKey="balances" childKey="symbol" type="string" required={false}>
symbol field
</NestedParam>
<NestedParam parentKey="balances" childKey="balance" type="string" required={false}>
balance field
</NestedParam>
<NestedParam parentKey="balances" childKey="decimals" type="number" required={false}>
decimals field
</NestedParam>

  </Expandable>
</ResponseField>

import { Turnkey } from "@turnkey/sdk-server";
