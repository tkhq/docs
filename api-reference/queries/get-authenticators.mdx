<ResponseField name="authenticators" type="array" required={true}>
  A list of authenticators.
  <Expandable title="authenticators details">

</NestedParam>

      </Expandable>
    </NestedParam>

attestationType field

</NestedParam>

</NestedParam>

      credential field

    </NestedParam>

<RequestExample>

```bash cURL
curl --request POST \
  --url https://api.turnkey.com/public/v1/submit//public/v1/query/get_authenticators \
  --header 'Accept: application/json' \
  --header 'Content-Type: application/json' \
  --header "X-Stamp: <YOUR_API_KEY.YOUR_API_SECRET>" \
  --data '{
    "type": "ACTIVITY_TYPE_UNKNOWN_V1",
    "timestampMs": "<string> (e.g., 1745002503362)",
    "organizationId": "<string> (Your Organization ID)",
    "parameters": {}
}'
```
