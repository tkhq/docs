        <NestedParam parentKey="mfaPolicy.requiredAuthenticationMethods" childKey="any" type="array" required={true}>

      <Expandable title="any details">
        <NestedParam parentKey="mfaPolicy.requiredAuthenticationMethods.any" childKey="type" type="enum" required={true}>

</NestedParam>
