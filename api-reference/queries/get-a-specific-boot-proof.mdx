---

import { Authorizations } from "/snippets/api/authorizations.mdx";
import { H3Bordered } from "/snippets/h3-bordered.mdx";

<EndpointPath type="query" path="get_boot_proof" />

<Authorizations />

<H3Bordered text="Body" />

<ParamField body="organizationId" type="string" required={true}>

Unique identifier for a given Organization.
</ParamField>

<ParamField body="ephemeralKey" type="string" required={true}>

Hex encoded ephemeral public key.
</ParamField>


<H3Bordered text="Response" />
A successful response returns the following fields:

<ResponseField name="bootProof" type="object" required={true}>
  bootProof field
  <Expandable title="bootProof details">
    <NestedParam parentKey="bootProof" childKey="ephemeralPublicKeyHex" type="string" required={true}>
The hex encoded Ephemeral Public Key.
</NestedParam>
<NestedParam parentKey="bootProof" childKey="awsAttestationDocB64" type="string" required={true}>
The DER encoded COSE Sign1 struct Attestation doc.
</NestedParam>
<NestedParam parentKey="bootProof" childKey="qosManifestB64" type="string" required={true}>
The borsch serialized base64 encoded Manifest.
</NestedParam>
<NestedParam parentKey="bootProof" childKey="qosManifestEnvelopeB64" type="string" required={true}>
The borsch serialized base64 encoded Manifest Envelope.
</NestedParam>
<NestedParam parentKey="bootProof" childKey="deploymentLabel" type="string" required={true}>
The label under which the enclave app was deployed.
</NestedParam>
<NestedParam parentKey="bootProof" childKey="enclaveApp" type="string" required={true}>
Name of the enclave app
</NestedParam>
<NestedParam parentKey="bootProof" childKey="owner" type="string" required={true}>
Owner of the app i.e. 'tkhq'
</NestedParam>
<NestedParam parentKey="bootProof" childKey="createdAt" type="object" required={true}>
      createdAt field
      <Expandable title="createdAt details">
        <NestedParam parentKey="bootProof.createdAt" childKey="seconds" type="string" required={true}>
seconds field
</NestedParam>
<NestedParam parentKey="bootProof.createdAt" childKey="nanos" type="string" required={true}>
nanos field
</NestedParam>

      </Expandable>
    </NestedParam>

  </Expandable>
</ResponseField>

<RequestExample>

```bash title="cURL"
curl --request POST \
  --url https://api.turnkey.com/public/v1/query/get_boot_proof \
  --header 'Accept: application/json' \
  --header 'Content-Type: application/json' \
  --header "X-Stamp: <string> (see Authorizations)" \
  --data '{
    "organizationId": "<string>",
    "ephemeralKey": "<string>"
}'
```

```javascript title="JavaScript"
import { Turnkey } from "@turnkey/sdk-server";

const turnkeyClient = new Turnkey({
  apiBaseUrl: "https://api.turnkey.com",
  apiPublicKey: process.env.API_PUBLIC_KEY!,
  apiPrivateKey: process.env.API_PRIVATE_KEY!,
  defaultOrganizationId: process.env.ORGANIZATION_ID!,
});

const response = await turnkeyClient.apiClient().getBootProof({
  organizationId: "<string> (Unique identifier for a given Organization.)",
  ephemeralKey: "<string> (Hex encoded ephemeral public key.)"
});
```
