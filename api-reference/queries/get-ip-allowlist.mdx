---
title: "Get IP Allowlist"
description: "Get IP allowlist and rules for an organization."
---

import { Authorizations } from "/snippets/api/authorizations.mdx";
import { H3Bordered } from "/snippets/h3-bordered.mdx";
import { NestedParam } from "/snippets/nested-param.mdx";
import { EndpointPath } from "/snippets/api/endpoint.mdx";

<EndpointPath type="query" path="get_ip_allowlist" />

<Authorizations />

<H3Bordered text="Body" />

<ParamField body="organizationId" type="string" required={true}>

Unique identifier for a given organization.
</ParamField>

<ParamField body="publicKey" type="string" required={false}>

If provided, return only the allowlist for this specific API key. If omitted, all allowlists for the organization are returned.
</ParamField>


<H3Bordered text="Response" />
A successful response returns the following fields:

<ResponseField name="allowlist" type="object" required={true}>
  allowlist field
  <Expandable title="allowlist details">
    <NestedParam parentKey="allowlist" childKey="organizationId" type="string" required={true}>
Unique identifier for the organization this allowlist belongs to.
</NestedParam>
<NestedParam parentKey="allowlist" childKey="rules" type="array" required={true}>
      List of IP allowlist rules with their metadata.
      <Expandable title="rules details">
        <NestedParam parentKey="allowlist.rules" childKey="cidr" type="string" required={true}>
CIDR block (e.g., '192.168.1.0/24').
</NestedParam>
<NestedParam parentKey="allowlist.rules" childKey="label" type="string" required={false}>
Optional human-readable label for this rule.
</NestedParam>
<NestedParam parentKey="allowlist.rules" childKey="createdAt" type="string" required={false}>
Creation timestamp as millisecond epoch string.
</NestedParam>

      </Expandable>
    </NestedParam>
<NestedParam parentKey="allowlist" childKey="publicKey" type="string" required={false}>
Public key of the API key this allowlist applies to. Null means the allowlist applies to the entire organization.
</NestedParam>
<NestedParam parentKey="allowlist" childKey="enabled" type="boolean" required={false}>
Whether the IP allowlist is enabled. Only present for organization-level allowlists. Null for API key-level allowlists (presence of the allowlist implies enablement).
</NestedParam>
<NestedParam parentKey="allowlist" childKey="onEvaluationError" type="string" required={false}>
Behavior when an error occurs during IP allowlist evaluation. Valid values: ALLOW, DENY. Defaults to DENY.
</NestedParam>

  </Expandable>
</ResponseField>

<RequestExample>

```bash title="cURL"
curl --request POST \
  --url https://api.turnkey.com/public/v1/query/get_ip_allowlist \
  --header 'Accept: application/json' \
  --header 'Content-Type: application/json' \
  --header "X-Stamp: <string> (see Authorizations)" \
  --data '{
    "organizationId": "<string>",
    "publicKey": "<string>"
}'
```

```javascript title="JavaScript"

  organizationId: "<string> (Unique identifier for a given organization.)",

});
```
