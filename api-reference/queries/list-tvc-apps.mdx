<ResponseField name="tvcApps" type="array" required={true}>
  A list of TVC Apps.
  <Expandable title="tvcApps details">
    <NestedParam parentKey="tvcApps" childKey="id" type="string" required={true}>
Unique Identifier for this TVC App.
</NestedParam>
<NestedParam parentKey="tvcApps" childKey="organizationId" type="string" required={true}>
Unique Identifier of the Organization for this TVC App
</NestedParam>

Name for this TVC App.
</NestedParam>

Public key for the Quorum Key associated with this TVC App
</NestedParam>
<NestedParam parentKey="tvcApps" childKey="manifestSetId" type="string" required={true}>
Unique Identifier of the Manifest Set (people who can approve manifests)
</NestedParam>

Unique Identifier of the Share Set (people who have a share of the Quorum Key)
</NestedParam>
