---
title: "Spark prepare transfer"
description: "Atomically compute key tweaks, Feldman-split them for Spark Operators, encrypt the receiver's new leaf key, and sign the transfer payload — all inside the Turnkey enclave."
---

<ParamField body="parameters" type="object" required={true} path="parameters">
  <p>The parameters object containing the specific intent data for this activity.</p>
  <Expandable title="details">
    <NestedParam parentKey="parameters" childKey="signWith" type="string" required={true} default="">
    The Spark address of the sender's signing identity. Determines which enclave-held identity key is used to derive leaf keys and sign the transfer payload.
    </NestedParam>

    <ParamField body="transfer" type="object" required={true} path="parameters.transfer">
      <p>The complete transfer descriptor including the receiver, Spark Operators, and all leaves being transferred.</p>
      <Expandable title="transfer details">
        <NestedParam parentKey="parameters.transfer" childKey="transferId" type="string" required={true} default="">
          A UUID v7 that uniquely identifies this transfer. Used by the enclave and Spark Operators to correlate all messages in the transfer flow.
        </NestedParam>
        <NestedParam parentKey="parameters.transfer" childKey="threshold" type="number" required={true} default="">
          The Shamir secret sharing threshold — the minimum number of Spark Operators that must cooperate to reconstruct any key tweak share. Must be ≤ the number of `operatorRecipients`.
        </NestedParam>
        <NestedParam parentKey="parameters.transfer" childKey="receiverPublicKey" type="string" required={true} default="">
          The receiver's Spark identity public key, hex-encoded (compressed secp256k1). The enclave ECIES-encrypts the receiver's new leaf private key to this public key so only the receiver can decrypt it during the claim step.
        </NestedParam>
        <ParamField body="operatorRecipients" type="array" required={true} path="parameters.transfer.operatorRecipients">
          <p>The list of Spark Operators that will receive encrypted key tweak shares. Each operator's Shamir share is ECIES-encrypted to its individual encryption public key.</p>
          <Expandable title="item details">
            <NestedParam parentKey="parameters.transfer.operatorRecipients" childKey="operatorId" type="string" required={true} default="">
              The Spark Operator identifier (e.g. `"SO1"`).
            </NestedParam>
            <NestedParam parentKey="parameters.transfer.operatorRecipients" childKey="encryptionPublicKey" type="string" required={true} default="">
              The operator's ECIES encryption public key, hex-encoded. The enclave uses this key to encrypt the operator's Shamir share so only that operator can decrypt it.
            </NestedParam>
          </Expandable>
        </ParamField>
        <ParamField body="leaves" type="array" required={true} path="parameters.transfer.leaves">
          <p>The list of Spark leaves being transferred. Each entry specifies the old leaf key, the new leaf key, and the pre-aggregated refund signatures that were produced in the preceding `SPARK_SIGN_FROST` call.</p>
          <Expandable title="item details">
            <NestedParam parentKey="parameters.transfer.leaves" childKey="leafId" type="string" required={true} default="">
              The identifier of the leaf being transferred.
            </NestedParam>
            <NestedParam parentKey="parameters.transfer.leaves" childKey="oldLeafDerivation" type="object" required={true} default="">
              Derivation descriptor for the sender's current leaf key. Must use `signingLeaf` with the current leaf's `leafId` — e.g. `{ "signingLeaf": { "leafId": "<old-leaf-id>" } }`. The enclave derives the current leaf private key to compute the tweak.
            </NestedParam>
            <NestedParam parentKey="parameters.transfer.leaves" childKey="newLeafDerivation" type="object" required={true} default="">
              Derivation descriptor for the receiver's new leaf key. Must use `signingLeaf` with the new leaf's `leafId` — e.g. `{ "signingLeaf": { "leafId": "<new-leaf-id>" } }`. The enclave derives this key and encrypts it to the receiver's identity public key.
            </NestedParam>
            <NestedParam parentKey="parameters.transfer.leaves" childKey="refundSignature" type="string" required={true} default="">
              The fully aggregated Schnorr signature for the new leaf's CPFP refund transaction, hex-encoded. Produced by aggregating the partial signatures from `SPARK_SIGN_FROST` with the Spark Operators' shares.
            </NestedParam>
            <NestedParam parentKey="parameters.transfer.leaves" childKey="directRefundSignature" type="string" required={true} default="">
              The fully aggregated Schnorr signature for the new leaf's direct refund transaction, hex-encoded.
            </NestedParam>
            <NestedParam parentKey="parameters.transfer.leaves" childKey="directFromCpfpRefundSignature" type="string" required={true} default="">
              The fully aggregated Schnorr signature for the new leaf's direct-from-CPFP refund transaction, hex-encoded.
            </NestedParam>
          </Expandable>
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
