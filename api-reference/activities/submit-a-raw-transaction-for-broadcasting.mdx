```bash title="cURL"
  --header "X-Stamp: <string> (see Authorizations)" \
  --data '{
    "type": "ACTIVITY_TYPE_ETH_SEND_RAW_TRANSACTION",

    "parameters": {
        "signedTransaction": "<string>",
        "caip2": "<eip155:1>"
    }
}'
```
