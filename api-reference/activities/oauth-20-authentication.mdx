const response = await turnkeyClient.apiClient().oauth2Authenticate({
  oauth2CredentialId: "<string> (The OAuth 2.0 credential id whose client_id and client_secret will be used in the OAuth 2.0 flow)",

  codeVerifier: "<string> (The code verifier used by OAuth 2.0 PKCE providers)",

  bearerTokenTargetPublicKey: "<string> (An optional P256 public key to which, if provided, the bearer token will be encrypted and returned via the `encrypted_bearer_token` claim of the OIDC Token)"
});
