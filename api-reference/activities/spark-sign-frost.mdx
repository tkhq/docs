<ParamField body="parameters" type="object" required={true} path="parameters">
  <p>The parameters object containing the specific intent data for this activity.</p>
  <Expandable title="details">
    <NestedParam parentKey="parameters" childKey="signWith" type="string" required={true} default="">
    The Spark address of the signing identity (e.g. `spark1pgss...`). Determines which enclave-held identity key is used to derive leaf signing keys.
    </NestedParam>

    <ParamField body="signatures" type="array" required={true} path="parameters.signatures">
      <p>A batch of one or more signature requests. A single call can sign multiple transactions (e.g. branch + exit at deposit time, or up to 6 refund transactions per transfer).</p>
      <Expandable title="item details">
        <NestedParam parentKey="parameters.signatures" childKey="derivation" type="object" required={true} default="">
          The key derivation descriptor that identifies which leaf key to use for this signature.
          <Expandable title="derivation details">
            <NestedParam parentKey="parameters.signatures.derivation" childKey="type" type="enum<string>" required={true} default="">
              Selects the key derivation strategy.

              | Value | Required Field | Used For |
              | --- | --- | --- |
              | `SPARK_KEY_TYPE_DEPOSIT` | _(none)_ | Single-use deposit transactions |
              | `SPARK_KEY_TYPE_STATIC_DEPOSIT_HD` | `index` | Static (reusable) deposit transactions |
              | `SPARK_KEY_TYPE_SIGNING_HD` | `leafId` | Active leaf signing — withdrawals, transfer refunds |

              Enum options: `SPARK_KEY_TYPE_DEPOSIT`, `SPARK_KEY_TYPE_STATIC_DEPOSIT_HD`, `SPARK_KEY_TYPE_SIGNING_HD`
            </NestedParam>
            <NestedParam parentKey="parameters.signatures.derivation" childKey="leafId" type="string" required={false} default="">
              Required when `type` is `SPARK_KEY_TYPE_SIGNING_HD`. Identifies the specific Spark leaf whose HD-derived signing key should be used.
            </NestedParam>
            <NestedParam parentKey="parameters.signatures.derivation" childKey="index" type="number" required={false} default="">
              Required when `type` is `SPARK_KEY_TYPE_STATIC_DEPOSIT_HD`. The integer index of the static deposit address (e.g. `0` for the first static deposit address at path `m/8797555'/{account}'/3'/0'`).
            </NestedParam>
          </Expandable>
        </NestedParam>
        <NestedParam parentKey="parameters.signatures" childKey="message" type="string" required={true} default="">
          The Bitcoin transaction sighash to sign, encoded as a hex string. Must match the sighash committed to in `operatorCommitments`.
        </NestedParam>
        <NestedParam parentKey="parameters.signatures" childKey="verifyingKey" type="string" required={true} default="">
          The aggregate Taproot public key (user key + Spark Operator key shares combined), hex-encoded. Used by the enclave to validate that the partial signature is consistent with the expected aggregate key.
        </NestedParam>
        <ParamField body="operatorCommitments" type="array" required={true} path="parameters.signatures.operatorCommitments">
          <p>The FROST nonce commitments provided by each participating Spark Operator for this signature session. Each operator must provide both a hiding and binding commitment before signing begins.</p>
          <Expandable title="item details">
            <NestedParam parentKey="parameters.signatures.operatorCommitments" childKey="id" type="string" required={true} default="">
              The Spark Operator identifier (e.g. `"SO1"`).
            </NestedParam>
            <NestedParam parentKey="parameters.signatures.operatorCommitments" childKey="hiding" type="string" required={true} default="">
              The operator's FROST hiding nonce commitment, hex-encoded.
            </NestedParam>
            <NestedParam parentKey="parameters.signatures.operatorCommitments" childKey="binding" type="string" required={true} default="">
              The operator's FROST binding nonce commitment, hex-encoded.
            </NestedParam>
          </Expandable>
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>
