<ParamField body="type" type="enum<string>" required={true}>

  <Expandable title="details">
    <NestedParam parentKey="parameters" childKey="from" type="string" required={true} default="">

    <NestedParam parentKey="parameters" childKey="sponsor" type="boolean" required={false} default="">

    <NestedParam parentKey="parameters" childKey="caip2" type="enum<string>" required={true} default="">

    </NestedParam>
    
    <NestedParam parentKey="parameters" childKey="to" type="string" required={true} default="">
    Recipient address as a hex string with 0x prefix.
    </NestedParam>
    
    <NestedParam parentKey="parameters" childKey="value" type="string" required={false} default="">
    Amount of native asset to send in wei.
    </NestedParam>
    
    <NestedParam parentKey="parameters" childKey="data" type="string" required={false} default="">
    Hex-encoded call data for contract interactions.
    </NestedParam>
    
    <NestedParam parentKey="parameters" childKey="nonce" type="string" required={true} default="">
    Transaction nonce.
    </NestedParam>
    
    <NestedParam parentKey="parameters" childKey="gasLimit" type="string" required={true} default="">
    Maximum amount of gas to use for this transaction.
    </NestedParam>
    
    <NestedParam parentKey="parameters" childKey="maxFeePerGas" type="string" required={true} default="">
    Maximum total fee per gas unit (base fee + priority fee) in wei, for EIP-1559 transactions.
    </NestedParam>
    
    <NestedParam parentKey="parameters" childKey="maxPriorityFeePerGas" type="string" required={true} default="">
    Maximum priority fee (tip) per gas unit in wei, for EIP-1559 transactions.
    </NestedParam>
    
    <NestedParam parentKey="parameters" childKey="deadline" type="string" required={false} default="">
    Unix timestamp after which the Gas Station meta-transaction is no longer valid. Only used when sponsor=true.
    </NestedParam>
  </Expandable>
